    // æ³¨å†Œè‡ªå®šä¹‰éªŒè¯å™¨

	if err != nil {
		c.JSON(400, gin.H{
			"error": err.Error(),
		})
		return
	}

	c.JSON(200, testJson)
}

func checkSchool(fl validator.FieldLevel) bool {
	if fl.Field().String() == "school" {
		return true
	}

	return false
