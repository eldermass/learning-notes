# 工厂模式

## 简单工厂模式

根据传入的静态值不同，产生不同的实例,创建的对象数量较少，对象的创建逻辑不复杂时使用。

```js
// 传入角色类型，获取对应角色的 user 对象
let userFactory = function (role) {
    function User(opts) {
        this.name = opts.name
        this.viewPage = opts.viewPage
    }
    switch (role) {
        case "superAdmin":
            return new User({
                name: "超级管理员",
                viewPage: ["首页", "通讯录", "发现页", "应用数据", "权限管理"],
            })
        case "admin":
            return new User({
                name: "管理员",
                viewPage: ["首页", "通讯录", "发现页", "应用数据"],
            })
        case "user":
            return new User({
                name: "用户",
                viewPage: ["首页", "通讯录", "发现页"],
            })
    }
}

// es6 写法
class Factory {
    constructor(opts) {
        this.name = opts.name
        this.viewPage = opts.viewPage
    }
    static getInstance(role) {
        switch (role) {
            case "superAdmin":
                return new Factory({
                    name: "超级管理员",
                    viewPage: [
                        "首页",
                        "通讯录",
                        "发现页",
                        "应用数据",
                        "权限管理",
                    ],
                })
            case "admin":
                return new Factory({
                    name: "管理员",
                    viewPage: ["首页", "通讯录", "发现页", "应用数据"],
                })
            case "user":
                return new Factory({
                    name: "用户",
                    viewPage: ["首页", "通讯录", "发现页"],
                })
        }
    }
}
```

## 工厂方法模式

```js
// 通过传入的角色获取原型上相应方法的实例
let userFactory = function (role) {
    // 保证实例化创建
    if (this instanceof userFactory) {
        return new this[role]()
    } else {
        return new userFactory(role)
    }
}

// 工厂方法函数的原型中设置所有对象的构造函数
// 添加新的角色，就只需要在原型里添加一个角色
userFactory.prototype = {
    superAdmin: function () {
        this.name = "超级管理员"
        this.viewPage = ["首页", "通讯录", "发现页", "应用数据", "权限管理"]
    },
    admin: function () {
        this.name = "管理员"
        this.viewPage = ["首页", "通讯录", "发现页", "应用数据"]
    },
    normalUser: function () {
        this.name = "普通用户"
        this.viewPage = ["首页", "通讯录", "发现页"]
    },
}
```

es6 里的写法

```js
class Factory {
    constructor(name = "", viewPage = []) {
        if (new.target === Factory) {
            throw new Error("抽象类不能实例化")
        }
        this.name = name
        this.viewPage = viewPage
    }
    getName() {
        return this.name
    }
}
class FacUser extends Factory {
    constructor(name, viewPage) {
        super(name, viewPage)
    }
    create(role) {
        switch (role) {
            case "superAdmin":
                return new FacUser("超级管理员", [
                    "首页",
                    "通讯录",
                    "发现页",
                    "应用数据",
                    "权限管理",
                ])
            case "admin":
                return new FacUser("管理员", [
                    "首页",
                    "通讯录",
                    "发现页",
                    "应用数据",
                ])
            case "user":
                return new FacUser("用户", ["首页", "通讯录", "发现页"])
        }
    }
}
```

## 抽象工厂模式

依靠继承来实现抽象方法的类簇管理

```js
// 抽象类
let accountAbstractFactory = function (subType, superType) {
    // subType是fn seperType是str
    if (typeof accountAbstractFactory[superType] === "function") {
        let F = function () {}
        F.prototype = new accountAbstractFactory[superType]()
        F.constructor = subType
        subType.prototype = new F()
    } else {
        throw new Error("抽象类不存在")
    }
}
// 微信抽象类
accountAbstractFactory.wechatUser = function () {
    this.type = "wechat"
}
accountAbstractFactory.wechatUser.prototype = {
    getName: function () {
        throw new Error("抽象方法不能调用")
    },
}
// 微博抽象类
accountAbstractFactory.weiboUser = function () {
    this.type = "weibo"
}
accountAbstractFactory.weiboUser.prototype = {
    getName: function () {
        throw new Error("抽象方法不能调用")
    },
}

// 普通微信用户子类
function wechat(name) {
    this.name = name
    this.viewPage = ["首页", "通讯录", "发现页"]
}
// 工厂方法继承超类
accountAbstractFactory(wechat, "wechatUser")
// 实现抽象方法
wechat.prototype.getName = function () {
    return this.name
}

// 普通微博用户子类
function weibo(name) {
    this.name = name
    this.viewPage = ["首页", "通讯录", "发现页"]
}
accountAbstractFactory(weibo, "weiboUser")
weibo.prototype.getName = function () {
    return this.name
}
```

es6 写法

```js
// 抽象用户类
class User {
    constructor(type) {
        if (new.target === User) {
            throw new Error("抽象类不能实例化!")
        }
        this.type = type
    }
}

class UserOfWechat extends User {
    constructor(name) {
        super("wechat")
        this.name = name
        this.viewPage = ["首页", "通讯录", "发现页"]
    }
}

class UserOfQq extends User {
    constructor(name) {
        super("qq")
        this.name = name
        this.viewPage = ["首页", "通讯录", "发现页"]
    }
}

class UserOfWeibo extends User {
    constructor(name) {
        super("weibo")
        this.name = name
        this.viewPage = ["首页", "通讯录", "发现页"]
    }
}

function getAbstractUserFactory(type) {
    switch (type) {
        case "wechat":
            return UserOfWechat
        case "qq":
            return UserOfQq
        case "weibo":
            return UserOfWeibo
        default:
            throw new Error("参数错误, 可选参数:superAdmin、admin、user")
    }
}

let WechatUserClass = getAbstractUserFactory("wechat")、
let wechatUser = new WechatUserClass("微信小李")、
```
