# 装饰器

import time
def decorator(func):
    # args 参数列表，kw 关键字参数(以字典接收剩余参数)
    def wrapper(*args, **kw):
        print(time.time())
        func(*args, **kw)
    return wrapper
# 使用装饰器

def f1():
    print('this is a func')
